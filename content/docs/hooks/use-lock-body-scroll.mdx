## Usage

```tsx {3, 6}
"use client"

import { useLockBodyScroll } from "@/hooks/use-lock-body-scroll"

export function Component() {
  useLockBodyScroll()

  return (
    <div style={{ height: "200vh" }}>
      <p>Scrolling is disabled</p>
    </div>
  )
}
```
