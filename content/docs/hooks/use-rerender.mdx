## Usage

```tsx {3, 6}
"use client"

import { useRerender } from "@/hooks/use-rerender"

export function Component() {
  const rerender = useRerender()
  console.log("Component rendered at:", new Date().toISOString())

  return (
    <div>
      <h1>Force Re-render Example</h1>
      <button onClick={rerender}>Re-render Component</button>
    </div>
  )
}
```

## API Reference

### Return Values

| Type         | Description                                                               |
| ------------ | ------------------------------------------------------------------------- |
| `() => void` | A memoized function that, when called, forces the component to re-render. |
