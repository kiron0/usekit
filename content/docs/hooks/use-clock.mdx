## Usage

```tsx {3, 6-14}
"use client"

import { useClock } from "@/hooks/use-clock"

export function Component() {
  const time = useClock({
    locale: "en-US",
    formatOptions: {
      hour: "2-digit",
      minute: "2-digit",
      second: "2-digit",
      hour12: true,
    },
  })

  return (
    <div>
      <p>Current time: {time}</p>
    </div>
  )
}
```

## API Reference

### Return Values

| Name   | Type     | Description                               |
| ------ | -------- | ----------------------------------------- |
| `time` | `string` | The current time formatted as `HH:mm:ss`. |
